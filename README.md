# 基于element代码补全

目前所支持的代码提示：

dlv <!-- 主框架 -->

elbutton <!-- 按钮 -->

elcheckbox <!-- 多选框 -->

elcheckboxg <!-- 多选组 -->

elrow <!-- 行 -->

elcol <!-- 列 -->

elform <!-- 表单 -->

elinput <!-- 输入框 -->

elinumber <!-- 计数器 -->

elpager <!-- 分页 -->

elradio <!-- 单选按钮 -->

elradiog <!-- 单选按钮组 -->

elselect <!-- 选择框 -->

elslider <!-- 滑动条 -->

elswitch <!-- 滑动开关 -->

eltable <!-- 表格 -->

elbread <!-- 面包导航 -->

elbreadi <!-- el-breadcrumb-item -->

eldialog <!-- 基础弹窗 -->

httppo <!-- post请求 -->
